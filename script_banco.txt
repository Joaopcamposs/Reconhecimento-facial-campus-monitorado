create database iftm;
use iftm;

create table cameras(
    id_da_camera int auto_increment primary key,
    usuario varchar(50),
    ip_da_camera varchar(50),
    senha varchar(50),
    estado varchar(50)
);

create table pessoas(
    id_pessoa int auto_increment primary key,
    nome varchar(50)
);

create table controle_captura(
    id_captura int,
    salvar_foto int
);
